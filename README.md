var n = new Date();
var y = n.getFullYear();
var m = n.getMonth() + 1;
var d = n.getDate();
var date = monthNames[n.getMonth()] + ' ' + d + ', ' + y;
var setDate = setInterval(function () {
  if (typeof $ === 'undefined') return;
  var $date = $('.app #date')
  if($date.length > 0) {
    $date.val(date).trigger('change');
    clearInterval(setDate);
  }
}, 500);
